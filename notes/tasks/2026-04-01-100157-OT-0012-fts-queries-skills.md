### 3. queries.ts -- searchRawByTool() function

Add a new section after the `hybridSearchRaw` function:

```typescript
// ─── Tool-based raw search ──────────────────────────────────

export interface RawToolSearchResult {
  rawId: number;
  sessionId: string | null;
  date: string;
  skill: string | null;
  summary: string | null;
  toolName: string;
  toolUseCount: number;
}

/**
 * Find raw sessions that used a specific tool.
 * Groups by raw_files to show one row per session with tool use count.
 *
 * @param toolName - exact tool name (e.g. "Read", "Bash", "Write") or FTS query
 * @param opts.limit - max results (default 20)
 * @param opts.dateFrom - filter by date >= (YYYY-MM-DD)
 * @param opts.dateTo - filter by date <= (YYYY-MM-DD)
 */
export function searchRawByTool(
  toolName: string,
  opts?: { limit?: number; dateFrom?: string; dateTo?: string }
): RawToolSearchResult[] {
  const raw = getRawDb();
  const limit = opts?.limit ?? 20;

  let sql = `
    SELECT rf.id as raw_id, rf.session_id, rf.date, rf.skill, rf.summary,
           rtc.tool_name, COUNT(*) as tool_use_count
    FROM raw_tool_calls rtc
    JOIN raw_files rf ON rf.id = rtc.raw_id
    WHERE rtc.tool_name = ?
  `;
  const params: any[] = [toolName];

  if (opts?.dateFrom) {
    sql += " AND rf.date >= ?";
    params.push(opts.dateFrom);
  }
  if (opts?.dateTo) {
    sql += " AND rf.date <= ?";
    params.push(opts.dateTo);
  }

  sql += " GROUP BY rf.id ORDER BY rf.date DESC LIMIT ?";
  params.push(limit);

  const rows = raw.prepare(sql).all(...params) as any[];
  return rows.map(r => ({
    rawId: r.raw_id,
    sessionId: r.session_id,
    date: r.date,
    skill: r.skill,
    summary: r.summary,
    toolName: r.tool_name,
    toolUseCount: r.tool_use_count,
  }));
}
```

## What This Does

`/bye` is the session closer -- the counterpart to `/morning`. Triggers reflection, syncs brain.db, consolidates memory, and logs to the daily note.

Raw context capture is handled automatically by the Stop hook (`capture-raw.ts`). No manual capture needed.

For formal dev sessions with task management, use `/dev-end` instead.

**Freddie handles everything inline.** No sub-agents.

## Deprecated

Raw session context is now captured automatically by the `capture-raw.ts` hook:
- **PostToolUse** -- appends each tool event incrementally (crash-safe)
- **PreCompact** -- copies full transcript before compaction
- **Stop** -- copies transcript, generates summary, indexes to brain.db

**You do not need to call `/save-raw` manually.** The Stop hook runs automatically when a session ends.

If you need to force a re-index of raw files, use:
```bash
bun src/tools/brain.ts --check
```

If you need to re-sync all raw files from scratch:
```bash
bun src/tools/brain.ts --sync-raw --full
